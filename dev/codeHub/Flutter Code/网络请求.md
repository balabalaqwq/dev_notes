import 'dart:convert' as dartConvert;

class CustomerCollector {
  /// 地址
  String address;

  ///标题
  String title;
  double latitude;
  double longitude;

  /// 城市名
  String cityName;

  /// 城市编码
  String cityCode;

  /// 省份名称
  String provinceName;

  /// 省份编码
  String provinceCode;

  /// 电话
  String tel;

  /// 兴趣点id
  String poiId;

  /// 商业区
  String businessArea;

  /// 距离
  int distance;

  /// 行政区划名称
  String adName;

  /// 行政区划编号
  String adCode;

  /// 采集类型
  String collectionType;

  /// 搜索关键字
  String keywords;

  /// 来源
  String dataSource;

  /// 批次名称
  String collectName;

  /// 批次创建时间
  String collectTime;

  /// 采集地点
  String collectAddress;

  String business;
  String collectDeviceCode;
  String taskId = "0";
  CustomerCollector({
    String address,
    String title,
    double latitude,
    double longitude,
    String cityName,
    String cityCode,
    String provinceName,
    String provinceCode,
    String tel,
    String poiId,
    String businessArea,
    int distance,
    String adName,
    String adCode,
    String collectionType,
    String keywords,
    String dataSource,
    String collectName,
    String collectTime,
    String collectAddress,
    String business,
    String collectDeviceCode,
  }) {
    this.address = address;
    this.title = title;
    this.latitude = latitude;
    this.longitude = longitude;
    this.cityName = cityName;
    this.cityCode = cityCode;
    this.provinceName = provinceName;
    this.provinceCode = provinceCode;
    this.tel = tel;
    this.poiId = poiId;
    this.businessArea = businessArea;
    this.distance = distance;
    this.adName = adName;
    this.adCode = adCode;
    this.collectionType = collectionType;
    this.keywords = keywords;
    this.dataSource = dataSource;
    this.collectName = collectName;
    this.collectTime = collectTime;
    this.collectAddress = collectAddress;
    this.business = business;
    this.collectDeviceCode = collectDeviceCode;
  }

  @override
  String toString() {
    return dartConvert.json.encode(this);
  }

  @override
  Map<String, dynamic> toJson() => {
        'address': address,
        'title': title,
        'latitude': latitude,
        'longitude': longitude,
        'cityName': cityName,
        'cityCode': cityCode,
        'provinceName': provinceName,
        'provinceCode': provinceCode,
        'tel': tel,
        'poiId': poiId,
        'businessArea': businessArea,
        'distance': distance,
        'adName': adName,
        'adCode': adCode,
        'collectionType': collectionType,
        'keywords': keywords,
        'dataSource': dataSource,
        'collectName': collectName,
        'collectTime': collectTime,
        'collectAddress': collectAddress,
        'business': business,
        'collectDeviceCode': collectDeviceCode,
        'taskId': "0"
      };
}
