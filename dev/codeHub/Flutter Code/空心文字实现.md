import 'package:flutter/material.dart';

final Color darkBlue = Color.fromARGB(255, 18, 32, 47);

void main() {
  runApp(MyApp());
}

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      theme: ThemeData.dark().copyWith(scaffoldBackgroundColor: darkBlue),
      debugShowCheckedModeBanner: false,
      home: Scaffold(
        backgroundColor: Colors.grey,
        body: Center(
          child: HollowedText(
            "aekyQKBM",
            100.0,
            Colors.blue,
          ),
        ),
      ),
    );
  }
}

class HollowedText extends StatelessWidget {
  final String text;
  final double size;
  final Color color;
  final Color color2;

  HollowedText(
    this.text,
    this.size,
    this.color, {
    this.color2 = Colors.transparent,
  });

  @override
  Widget build(BuildContext context) {
    return Stack(
      children: <Widget>[
        Align(
          alignment: Alignment(0, 0),
          child: Text(
            this.text,
            style: TextStyle(
              fontSize: this.size,
              foreground: Paint()
                ..style = PaintingStyle.stroke
                ..strokeWidth = 2
                ..color = this.color,
            ),
          ),
        ),
        Align(
          alignment: Alignment(0, 0),
          child: Text(
            this.text,
            style: TextStyle(
              fontSize: this.size,
              color: this.color2,
            ),
          ),
        ),
      ],
    );
  }
}
